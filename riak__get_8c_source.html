34 riak_error
35 riak_get_request_encode(riak_operation *rop,
36  riak_binary *bucket,
37  riak_binary *key,
38  riak_get_options *get_options,
39  riak_pb_message **req) {
40 
41  riak_config *cfg = riak_operation_get_config(rop);
42  RpbGetReq getmsg = RPB_GET_REQ__INIT;
43 
44  riak_operation_set_bucket(rop, bucket);
45  riak_operation_set_key(rop, key);
46  riak_binary_copy_to_pb(&getmsg.bucket, bucket);
47  riak_binary_copy_to_pb(&getmsg.key, key);
48 
49  // process get options
50  if(get_options != NULL) {
51  getmsg.has_r = get_options->has_r;
52  getmsg.r = get_options->r;
53  getmsg.has_pr = get_options->has_pr;
54  getmsg.pr = get_options->pr;
55  getmsg.has_basic_quorum = get_options->has_basic_quorum;
56  getmsg.basic_quorum = get_options->basic_quorum;
57  getmsg.has_notfound_ok = get_options->has_notfound_ok;
58  getmsg.notfound_ok = get_options->notfound_ok;
59  if (get_options->has_if_modified) {
60  getmsg.has_if_modified = get_options->has_if_modified;
61  riak_binary_copy_to_pb(&getmsg.if_modified, get_options->if_modified);
62  }
63  getmsg.has_head = get_options->has_head;
64  getmsg.head = get_options->head;
65  getmsg.has_deletedvclock = get_options->has_deletedvclock;
66  getmsg.deletedvclock = get_options->deletedvclock;
67  getmsg.has_timeout = get_options->has_timeout;
68  getmsg.timeout = get_options->timeout;
69  getmsg.has_sloppy_quorum = get_options->has_sloppy_quorum;
70  getmsg.sloppy_quorum = get_options->sloppy_quorum;
71  getmsg.has_n_val = get_options->has_n_val;
72  getmsg.n_val = get_options->n_val;
73  }
74  riak_uint32_t msglen = rpb_get_req__get_packed_size (&getmsg);
75  riak_uint8_t* msgbuf = (riak_uint8_t*)(cfg->malloc_fn)(msglen);
76  if (msgbuf == NULL) {
77  return ERIAK_OUT_OF_MEMORY;
78  }
79  rpb_get_req__pack (&getmsg, msgbuf);
80  riak_pb_message* request = riak_pb_message_new(cfg, MSG_RPBGETREQ, msglen, msgbuf);
81  if (request == NULL) {
82  return ERIAK_OUT_OF_MEMORY;
83  }
84  *req = request;
85  riak_operation_set_response_decoder(rop, (riak_response_decoder)riak_get_response_decode);
86 
87  return ERIAK_OK;
88 }

90 riak_error
91 riak_get_response_decode(riak_operation *rop,
92  riak_pb_message *pbresp,
93  riak_get_response **resp,
94  riak_boolean_t *done) {
95  // decode the PB response etc
96  riak_config *cfg = riak_operation_get_config(rop);
97  RpbGetResp *rpbresp = rpb_get_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
98  if (rpbresp == NULL) {
99  return ERIAK_MESSAGE_FORMAT;
100  }
101  riak_connection *cxn = riak_operation_get_connection(rop);
102  riak_log_debug(cxn, "riak_decode_get_response len=%d/pb unpack = 0x%lx\n", pbresp->len, (long)(rpbresp));
103  *done = RIAK_TRUE;
104  if (rpbresp == NULL) {
105  return ERIAK_OUT_OF_MEMORY;
106  }
107  int i = 0;
108  riak_get_response *response = (riak_get_response*)(cfg->malloc_fn)(sizeof(riak_get_response));
109  if (response == NULL) {
110  return ERIAK_OUT_OF_MEMORY;
111  }
112  memset(response, '\0', sizeof(riak_get_response));
113  response->_internal = rpbresp;
114 
115  if (rpbresp->has_vclock) {
116  response->has_vclock = RIAK_TRUE;
117  response->vclock = riak_binary_copy_from_pb(cfg, &(rpbresp->vclock));
118  if (response->vclock == NULL) {
119  riak_free(cfg, &response);
120  return ERIAK_OUT_OF_MEMORY;
121  }
122  }
123  if (rpbresp->has_unchanged) {
124  response->has_unmodified = RIAK_TRUE;
125  response->unmodified = rpbresp->unchanged;
126  }
127  if (rpbresp->n_content > 0) {
128  riak_error err = riak_object_new_array(cfg, &(response->content), rpbresp->n_content);
129  if (err != ERIAK_OK) {
130  riak_free(cfg, &response);
131  return 1;
132  }
133  response->n_content = rpbresp->n_content;
134  for(i = 0; i < rpbresp->n_content; i++) {
135  err = riak_object_new_from_pb(cfg, &(response->content[i]), rpbresp->content[i]);
136  // If any object allocation fails, clean up all previously allocated ones
137  if (err != ERIAK_OK) {
138  riak_object_free_array(cfg, &(response->content), i);
139  riak_free(cfg, &response);
140  return err;
141  }
142  response->content[i]->bucket = riak_binary_copy(cfg, riak_operation_get_bucket(rop));
143  response->content[i]->key = riak_binary_copy(cfg, riak_operation_get_key(rop));
144  response->content[i]->has_key = RIAK_TRUE;
145  }
146  }
147  *resp = response;
148 
149  return ERIAK_OK;
150 }

152 int
153 riak_print_get_response(riak_get_response *response,
154  char *target,
155  riak_size_t len) {
156  riak_int32_t left_to_write = len;
157  riak_int32_t wrote;
158  riak_print_binary_hex("V-Clock", response->vclock, &target, &wrote, &left_to_write);
159  riak_print_bool("Unmodified", response->unmodified, &target, &wrote, &left_to_write);
160  riak_print_bool("Deleted", response->deleted, &target, &wrote, &left_to_write);
161  riak_print_int("Objects", response->n_content, &target, &wrote, &left_to_write);
162 
163  riak_uint32_t i;
164  for(i = 0; (i < response->n_content) && (left_to_write > 0); i++) {
165  wrote = riak_object_print(response->content[i], target, left_to_write);
166  left_to_write -= wrote;
167  target += wrote;
168  }
169  return len-left_to_write;
170 }

172 void
173 riak_get_response_free(riak_config *cfg,
174  riak_get_response **resp) {
175  riak_get_response *response = *resp;
176  if (response == NULL) return;
177  if (response->n_content > 0) {
178  riak_object_free_array(cfg, &(response->content), response->n_content);
179  }
180  riak_free(cfg, &(response->vclock));
181  rpb_get_resp__free_unpacked(response->_internal, cfg->pb_allocator);
182  riak_free(cfg, resp);
183 }
