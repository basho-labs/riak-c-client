42 void
43 riak_libevent_connection_cb(struct bufferevent *bev,
44  short events,
45  void *ptr) {
46  riak_connection *cxn = (riak_connection*)ptr;
47  if (events & BEV_EVENT_CONNECTED) {
48  riak_log_debug(cxn, "%s","Connect okay.");
49  } else if (events & (BEV_EVENT_ERROR|BEV_EVENT_EOF)) {
50  char *reason = "BEV_EVENT_ERROR";
51  if (events & BEV_EVENT_ERROR) {
52  reason = "BEV_EVENT_EOF";
53  int err = bufferevent_socket_get_dns_error(bev);
54  if (err)
55  riak_log_error(cxn, "DNS error: %s", evutil_gai_strerror(err));
56  }
57 #ifdef _RIAK_DEBUG
58  struct evbuffer *ev_read = bufferevent_get_input(bev);
59  struct evbuffer *ev_write = bufferevent_get_output(bev);
60 #endif
61  riak_log_debug(cxn, "Closing because of %s [read event=%p, write event=%p]",
62  reason, (void*)ev_read, (void*)ev_write);
63  bufferevent_free(bev);
64  event_base_loopexit(bufferevent_get_base(bev), NULL);
65  } else if (events & BEV_EVENT_TIMEOUT) {
66  riak_log_debug(cxn, "%s","Timeout Event");
67  bufferevent_free(bev);
68  event_base_loopexit(bufferevent_get_base(bev), NULL);
69  } else {
70  riak_log_debug(cxn, "Event %d", events);
71  }
72 }

148 riak_error
149 riak_libevent_new(riak_libevent **rev_target,
150  riak_operation *rop,
151  struct event_base *base) {
152  riak_config *cfg = riak_operation_get_config(rop);
153  riak_connection *cxn = riak_operation_get_connection(rop);
154 
155  riak_libevent *rev = (riak_libevent*)riak_config_clean_allocate(cfg, sizeof(riak_libevent));
156  if (rev == NULL) {
157  riak_log_critical_config(cfg, "%s", "Could not allocate a riak_libevent");
158  return ERIAK_OUT_OF_MEMORY;
159  }
160  *rev_target = rev;
161  rev->rop = rop;
162  rev->base = base;
163 
164  rev->bevent = bufferevent_socket_new(rev->base,
165  riak_connection_get_fd(cxn),
166  BEV_OPT_CLOSE_ON_FREE|BEV_OPT_DEFER_CALLBACKS);
167  if (rev->bevent == NULL) {
168  riak_log_critical_config(cfg,
169  "Could not create bufferevent [fd %d]",
170  riak_connection_get_fd(cxn));
171  return ERIAK_OUT_OF_MEMORY;
172  }
173  int enabled = bufferevent_enable(rev->bevent, EV_READ|EV_WRITE);
174  if (enabled != 0) {
175  riak_log_critical_config(cfg,
176  "Could not enable bufferevent [fd %d]",
177  riak_connection_get_fd(cxn));
178  return ERIAK_EVENT;
179  }
180 
181  // Set the internal read and write callbacks
182  bufferevent_setcb(rev->bevent,
183  riak_libevent_result_cb,
184  riak_libevent_write_event_cb,
185  riak_libevent_connection_cb,
186  rev);
187 
188  return ERIAK_OK;
189 }
