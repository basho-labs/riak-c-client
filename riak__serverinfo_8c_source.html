48 riak_serverinfo_response_decode(riak_operation *rop,
49  riak_pb_message *pbresp,
50  riak_serverinfo_response **resp,
51  riak_boolean_t *done) {
52  // decode the PB response etc
53  riak_config *cfg = riak_operation_get_config(rop);
54  RpbGetServerInfoResp *rpbresp = rpb_get_server_info_resp__unpack(cfg->pb_allocator, (pbresp->len)-1, (uint8_t*)((pbresp->data)+1));
55  if (rpbresp == NULL) {
56  return ERIAK_MESSAGE_FORMAT;
57  }
58  riak_connection *cxn = riak_operation_get_connection(rop);
59  riak_log_debug(cxn, "riak_decode_server_response len=%d/pb unpack = 0x%lx\n", pbresp->len, (long)(rpbresp));
60  *done = RIAK_TRUE;
61  if (rpbresp == NULL) {
62  return ERIAK_OUT_OF_MEMORY;
63  }
64  riak_serverinfo_response *response = (riak_serverinfo_response*)(cfg->malloc_fn)(sizeof(riak_serverinfo_response));
65  if (response == NULL) {
66  return ERIAK_OUT_OF_MEMORY;
67  }
68  memset(response, '\0', sizeof(riak_serverinfo_response));
69  response->_internal = rpbresp;
70 
71  if (rpbresp->has_node) {
72  response->has_node = RIAK_TRUE;
73  response->node = riak_binary_copy_from_pb(cfg, &(rpbresp->node));
74  if (response->node == NULL) {
75  return ERIAK_OUT_OF_MEMORY;
76  }
77  }
78  if (rpbresp->has_server_version) {
79  response->has_server_version = RIAK_TRUE;
80  response->server_version = riak_binary_copy_from_pb(cfg, &(rpbresp->server_version));
81  if (response->server_version == NULL) {
82  return ERIAK_OUT_OF_MEMORY;
83  }
84  }
85  *resp = response;
86 
87  return ERIAK_OK;
88 }

91 riak_serverinfo_response_print(riak_serverinfo_response *response,
92  char *target,
93  riak_size_t len) {
94  riak_int32_t left_to_write = len;
95  riak_int32_t wrote;
96  char buffer[1024];
97  if ((response->has_node) && (left_to_write > 0)) {
98  riak_binary_print(response->node, buffer, sizeof(buffer));
99  wrote = snprintf(target, left_to_write, "Node: %s\n", buffer);
100  left_to_write -= wrote;
101  target += wrote;
102  }
103  if ((response->has_server_version) && (left_to_write > 0)) {
104  riak_binary_print(response->server_version, buffer, sizeof(buffer));
105  wrote = snprintf(target, left_to_write, "Version: %s\n", buffer);
106  left_to_write -= wrote;
107  target += wrote;
108  }
109 }
